```python
def make_live_mode_json(tmp_path, enabled=True):
    data = {"enabled": enabled, "unlocked_at": "2026-04-17T09:00:00+08:00", "unlocked_by": "test"}
    (tmp_path / "live_mode.json").write_text(json.dumps(data))

    return {
        "symbol": "0050", "side": "buy", "quantity": 1000,

def make_mock_adapter(broker_order_id="ORD123", verify_result=None):
    adapter = MagicMock()
    submitted_order = Order(symbol="0050", action="buy", quantity=1000,
                            broker_order_id=broker_order_id, status="submitted")
    adapter._submit_order_impl = AsyncMock(return_value=submitted_order)
    if verify_result is None:
        verify_result = {"verified": True, "ghost": False,
                         "broker_order_id": broker_order_id, "polls": 1}
    adapter.verify_order_landed = AsyncMock(return_value=verify_result)
    return adapter
```

```python
def test_happy_path_order_written_to_orders_open(tmp_path):
    make_live_mode_json(tmp_path)
    adapter = make_mock_adapter("ORD001")
    result = asyncio.run(submit_live_order(make_valid_order(), adapter=adapter, state_dir=tmp_path))
    assert result["success"] is True
    assert result["verified"] is True
    assert result["ghost"] is False
    orders_open = json.loads((tmp_path / "orders_open.json").read_text())
    assert any(o["broker_order_id"] == "ORD001" for o in orders_open)
```

--- SCENARIO 2: Ghost order ---
```python
def test_ghost_order_logged_not_in_orders_open(tmp_path):
    make_live_mode_json(tmp_path)
    ghost_verify = {"verified": False, "ghost": True, "broker_order_id": "ORD002", "polls": 3}
    adapter = make_mock_adapter("ORD002", verify_result=ghost_verify)
    result = asyncio.run(submit_live_order(make_valid_order("ghost-001"), adapter=adapter, state_dir=tmp_path))
    assert result["success"] is False
    assert result["ghost"] is True
    # ghost_orders.jsonl must exist and contain the entry
    ghost_log = (tmp_path / "ghost_orders.jsonl")
    assert ghost_log.exists()
    record = json.loads(ghost_log.read_text().strip().splitlines()[-1])
    assert record["broker_order_id"] == "ORD002"
    # orders_open.json must NOT exist or must not contain this order
    if (tmp_path / "orders_open.json").exists():
        orders = json.loads((tmp_path / "orders_open.json").read_text())
        assert not any(o.get("broker_order_id") == "ORD002" for o in orders)
```

```python
def test_adapter_exception_graceful_error(tmp_path):
    make_live_mode_json(tmp_path)
    adapter = MagicMock()

    result = asyncio.run(submit_live_order(make_valid_order(), adapter=adapter, state_dir=tmp_path))
    assert result["success"] is False
    assert result["step"] == "submit"
```

```python
def test_double_submit_no_duplicate_in_orders_open(tmp_path):
    make_live_mode_json(tmp_path)
    adapter = make_mock_adapter("ORD003")
    order = make_valid_order("dup-001")
    asyncio.run(submit_live_order(order, adapter=adapter, state_dir=tmp_path))
    asyncio.run(submit_live_order(order, adapter=adapter, state_dir=tmp_path))
    orders_open = json.loads((tmp_path / "orders_open.json").read_text())
    entries = [o for o in orders_open if o.get("order_id") == "dup-001"]
    # At most one entry per order_id (idempotent or deduplicated)
    assert len(entries) <= 1
```
