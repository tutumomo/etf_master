```python
@app.get("/api/live-mode/status")
async def get_live_mode_status():
    """Live 模式授權狀態 — LIVE-02"""
    from scripts.etf_core import context as _ctx
    from scripts.etf_core.state_io import safe_load_json
    state_dir = _ctx.get_state_dir()
    live_mode = safe_load_json(state_dir / "live_mode.json", default={"enabled": False})
    backtest = safe_load_json(state_dir / "backtest_results.json", default={})
    quality_gate_passed = bool(backtest.get("quality_gate_passed", False))
    return {
        "enabled": bool(live_mode.get("enabled", False)),
        "quality_gate_passed": quality_gate_passed,
        "unlocked_at": live_mode.get("unlocked_at"),
        "backtest_summary": {
            "win_rate": backtest.get("win_rate"),
            "max_drawdown": backtest.get("max_drawdown"),
            "last_updated": backtest.get("last_updated")
        }
    }
```

```python
@app.post("/api/live-mode/unlock")
async def unlock_live_mode(req: LiveUnlockRequest):
    """Live 模式授權解鎖 — LIVE-02. 需要雙重確認字串 + 品質閘門通過."""
    from scripts.etf_core import context as _ctx
    from scripts.etf_core.state_io import safe_load_json, atomic_save_json
    from datetime import datetime
    from zoneinfo import ZoneInfo

    if req.confirm_1 != LIVE_CONFIRM_1 or req.confirm_2 != LIVE_CONFIRM_2:
        raise HTTPException(
            status_code=400,
            detail="Confirmation strings incorrect. "
                   f"Required: confirm_1='{LIVE_CONFIRM_1}', confirm_2='{LIVE_CONFIRM_2}'"
        )

    state_dir = _ctx.get_state_dir()
    backtest = safe_load_json(state_dir / "backtest_results.json", default={})
    if not backtest.get("quality_gate_passed", False):
        raise HTTPException(
            status_code=403,
            detail="Quality gate not passed. Run backtest_decision_outcomes.py first. "
                   "Required: win_rate >= 0.5 and max_drawdown <= 0.15."
        )

    live_mode = {
        "enabled": True,
        "unlocked_at": datetime.now(ZoneInfo("Asia/Taipei")).isoformat(),
        "unlocked_by": "dashboard",
        "quality_gate_passed_at_unlock": True
    }
    atomic_save_json(state_dir / "live_mode.json", live_mode)
    return {"success": True, "enabled": True, "unlocked_at": live_mode["unlocked_at"]}
```

```html
(function loadLiveModeStatus() {
  fetch('/api/live-mode/status')
    .then(function(r) { return r.json(); })
    .then(function(data) {
      var modeEl = document.getElementById('lm-mode-label');
      var gateEl = document.getElementById('lm-gate-label');
      var bsEl   = document.getElementById('lm-backtest-summary');
      var form   = document.getElementById('live-mode-gate-form');
      var btn    = document.getElementById('live-unlock-btn');
      var msgEl  = document.getElementById('live-gate-msg');

      // Use textContent exclusively — no user data injected into HTML
      modeEl.textContent = '目前模式：' + (data.enabled ? 'LIVE 模式' : 'Paper 模式');
      gateEl.textContent = '品質閘門：' + (data.quality_gate_passed ? '通過' : '未通過');

      var bs = data.backtest_summary || {};
      var wr = bs.win_rate != null ? (bs.win_rate * 100).toFixed(1) + '%' : 'N/A';
      var md = bs.max_drawdown != null ? (bs.max_drawdown * 100).toFixed(1) + '%' : 'N/A';
      bsEl.textContent = '回測：勝率 ' + wr + '，最大回撤 ' + md;

      if (!data.enabled) {
        form.style.display = 'block';
        if (data.quality_gate_passed) {
          btn.disabled = false;
          btn.style.opacity = '1';
          btn.style.cursor = 'pointer';
        } else {
          msgEl.textContent = '品質閘門未通過，無法解鎖';
        }
      }
    })
    .catch(function(e) {
      document.getElementById('lm-mode-label').textContent = '狀態載入失敗: ' + e;
    });
})();

function unlockLiveMode() {
  var c1 = document.getElementById('live-confirm-1').value;
  var c2 = document.getElementById('live-confirm-2').value;
  var msg = document.getElementById('live-gate-msg');
  msg.textContent = '提交中...';
  fetch('/api/live-mode/unlock', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({confirm_1: c1, confirm_2: c2})
  })
  .then(function(r) { return r.json().then(function(d) { return {status: r.status, data: d}; }); })
  .then(function(res) {
    if (res.status === 200) {
      msg.textContent = 'Live 模式已解鎖';
      msg.style.color = 'green';
      setTimeout(function() { location.reload(); }, 1500);
    } else {
      msg.textContent = '解鎖失敗: ' + (res.data.detail || '未知錯誤');
      msg.style.color = 'red';
    }
  })
  .catch(function(e) { msg.textContent = '錯誤: ' + e; msg.style.color = 'red'; });
}
</script>
```
